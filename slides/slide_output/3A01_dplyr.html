The Lahman data

Baseball stats

library(Lahman)
library(Lahman)
data(package = 'Lahman')

Your questions

  1. What is mean and median salary by year?
  2. Using the “Master” and “Salaries” tables, what is the min, max and mean salaries for all players who are (according to this data) alive.
  3. What is the ratio of the mean salary by year of left-handed batters to right-handed batters? Note: there are three kinds of batters in the database: “L”, “R” and “B”. Ignore ambidextrous (i.e. “B”) batters.

Answers - 1

library(tidyr)
library(dplyr)
dfSalaryByYear <- Salaries %>% 
  group_by(yearID) %>% 
  summarise(AvgSalary = mean(salary)
            , MedianSalary = median(salary))

Answers - 2

dfPlayerSalary <- Master %>% 
  filter(is.na(deathDate)) %>% 
  select(playerID, bats) %>% 
  inner_join(Salaries)

dfSalarySummary <- dfPlayerSalary %>% 
  summarise(MinSalary = min(salary)
            , MaxSalary = max(salary)
            , AvgSalary = mean(salary))

Answers - 3

dfYearAndHand <- dfPlayerSalary %>% 
  filter(bats != 'B') %>% 
  group_by(yearID, bats) %>% 
  summarise(AvgSalary = mean(salary)) %>% 
  spread(bats, AvgSalary) %>% 
  mutate(LeftRelativity = L / R)
